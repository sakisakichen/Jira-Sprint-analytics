# Bypasses Jira's 1,000-record limit using nextPageToken
def fetch_issues(jql):
    url = f"{JIRA_URL}rest/api/3/search/jql"
    all_issues, next_page_token = [], None

    while True:
        params = {"jql": jql, "maxResults": 50, "fields": FIELDS}
        if next_page_token:
            params["nextPageToken"] = next_page_token

        response = requests.get(url, headers=headers,
                            params=params, auth=auth, verify=False)
        data = response.json()
        all_issues.extend(data.get("issues", []))

        next_page_token = data.get("nextPageToken")
        if not next_page_token: break

    return all_issues

def extract_sprint_num(sprint_name):
    if pd.isna(sprint_name): return 9999
    matches = re.findall(r'T(\d+)', str(sprint_name))
    return int(matches[-1]) if matches else 9999
